     data/input/patient_001.mp4 \
     data/input/patient_001_scale.json

     data/input/patient_001.mp4 \

# ─── BALLOON TEST WORKFLOW ───────────────────────────────────────────

VIDEO=data/input/balloon_001.mp4

# 1. Scale calibration  (click 2 points on a known-size object, enter cm)
bash scripts/run_scale_picker.sh \
    "$VIDEO" \
    data/input/balloon_001_scale.json
# → Browser opens. Click frame_000 in sidebar, zoom in (wheel),
#   click 2 pixels on a ruler/box edge, enter real distance in cm,
#   click "Track to all frames", verify, close terminal with Ctrl+C.

# 2. Belly seed  (1 click on the balloon, propagated)
MODE=seed bash scripts/run_scale_picker.sh \
    "$VIDEO" \
    data/input/balloon_001_seed.json
# → Browser opens in seed mode. Click on the balloon center,
#   "Track seed to all frames", Ctrl+C.

# 3. Run the pipeline
SUBJECT=balloon \
SCALE_CALIB=data/input/balloon_001_scale.json \
SEED_POINTS=data/input/balloon_001_seed.json \
  bash scripts/run_belly_pipeline.sh \
    "$VIDEO" \
    data/output/balloon_001


# ─── PREGNANT WORKFLOW ───────────────────────────────────────────────

VIDEO=data/input/belly_001.mp4

# 1. Scale calibration (same as above)
bash scripts/run_scale_picker.sh \
    "$VIDEO" \
    data/input/belly_001_scale.json

# 2. Belly seed
MODE=seed bash scripts/run_scale_picker.sh \
    "$VIDEO" \
    data/input/belly_001_seed.json

SEED_POINTS=data/input/belly_001_seed.json \
  bash scripts/run_belly_pipeline.sh \
